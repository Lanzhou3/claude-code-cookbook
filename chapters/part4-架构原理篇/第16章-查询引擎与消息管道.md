## 16.3 异步生成器模式

### 16.3.1 流式响应

Claude Code 使用 `AsyncGenerator` 实现流式处理：

```typescript
// 异步生成器签名
async *submitMessage(
  messages: Message[],
  context: Context
): AsyncGenerator<SubmitMessageResult>
```

### 16.3.2 yield 机制

```typescript
// 伪代码示例
async *submitMessage(messages, context) {
  // 1. 构建请求
  const request = await buildRequest(messages, context)

  // 2. 发送 API 请求（流式）
  const stream = await api.sendMessage(request)

  // 3. 流式处理响应
  for await (const chunk of stream) {
    // 实时 yield，每收到一个 chunk 就输出
    yield {
      type: 'chunk',
      content: chunk
    }
  }

  // 4. 最终结果
  yield {
    type: 'complete',
    message: finalMessage
  }
}
```

### 16.3.3 使用方代码

```typescript
// 使用流式响应
for await (const result of queryEngine.submitMessage(messages, context)) {
  if (result.type === 'chunk') {
    process.stdout.write(result.content)
  } else if (result.type === 'complete') {
    console.log('\n对话完成')
  }
}
```

## 16.4 上下文构建

### 16.4.1 上下文组成

```typescript
interface Context {
  systemPrompt: SystemPrompt      // 系统提示词
  userContext: UserContext        // 用户上下文
  systemContext: SystemContext    // 系统上下文
  toolUseContext: ToolUseContext  // 工具使用上下文
}
```

### 16.4.2 上下文优先级

```
1. 系统提示 (System Prompt)
   - 工具定义
   - 行为规则
   - 输出格式

2. 项目上下文 (Project Context)
   - 语言/框架检测
   - 项目配置
   - 目录结构

3. 会话历史 (Session History)
   - 之前消息
   - 工具调用记录

4. 当前请求 (Current Request)
   - 用户最新输入
   - 临时状态
```

### 16.4.3 上下文注入

```typescript
const buildContext = async (
  baseMessages: Message[],
  context: Context
): Promise<Message[]> => {
  const result: Message[] = []

  // 1. 系统提示（固定）
  result.push({
    role: 'system',
    content: buildSystemPrompt(context)
  })

  // 2. 项目上下文（按需）
  if (context.systemContext.projectInfo) {
    result.push({
      role: 'system',
      content: buildProjectContext(context.systemContext.projectInfo)
    })
  }

  // 3. 会话历史（压缩后）
  result.push(...await compressHistory(baseMessages))

  return result
}
```


### 16.8.2 成本计算

```typescript
const calculateCost = (usage: TokenUsage, model: string): number => {
  const rates = {
    'claude-sonnet-4-6': {
      input: 3,      // $3 / M tokens
      output: 15     // $15 / M tokens
    },
    'claude-opus-4-6': {
      input: 15,
      output: 75
    }
  }

  const rate = rates[model]
  return (
    (usage.inputTokens * rate.input +
     usage.outputTokens * rate.output) / 1_000_000
  )
}
```

### 16.9.2 历史压缩

```typescript
// 当消息超过限制时压缩
const compressHistory = async (
  messages: Message[],
  maxMessages = 50
): Promise<Message[]> => {
  if (messages.length <= maxMessages) {
    return messages
  }

  // 保留最近的 N 条消息
  const recent = messages.slice(-maxMessages)

  // 将早期消息压缩为摘要
  const earlyMessages = messages.slice(0, -maxMessages)
  const summary = await generateSummary(earlyMessages)

  return [
    {
      role: 'system' as const,
      content: `[早期对话摘要]: ${summary}`
    },
    ...recent
  ]
}
```

## 16.10 中止与恢复

### 16.10.1 中止机制

```typescript
class QueryEngine {
  private abortController: AbortController = new AbortController()

  abort() {
    this.abortController.abort()
  }

  async *submitMessage(...) {
    try {
      for await (const chunk of stream) {
        // 检查是否被中止
        if (this.abortController.signal.aborted) {
          yield { type: 'aborted' }
          return
        }
        yield chunk
      }
    } catch (error) {
      if (error instanceofAbortError) {
        yield { type: 'aborted' }
      }
    }
  }
}
```

### 16.10.2 会话恢复

```typescript
// 从保存的状态恢复
const resume = async (

  const state = await loadSessionState(sessionId)

    ...state.config,
```

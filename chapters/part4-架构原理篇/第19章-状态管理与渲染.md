## 19.3 AppStateStore

### 19.3.1 响应式存储设计

```typescript
// state/AppStateStore.ts
class AppStateStore {
  private state: AppState
  private listeners: Set<Listener> = new Set()

  constructor(initialState: AppState) {
    this.state = initialState
  }

  // 获取状态
  getState(): AppState {
    return this.state
  }

  // 更新状态
  setState(updater: (prev: AppState) => AppState): void {
    const prevState = this.state
    const nextState = updater(prevState)

    // 只有状态真正变化时才通知
    if (nextState !== prevState) {
      this.state = nextState
      this.notifyListeners(nextState, prevState)
    }
  }

  // 订阅状态变化
  subscribe(listener: Listener): () => void {
    this.listeners.add(listener)
    return () => this.listeners.delete(listener)
  }

  private notifyListeners(next: AppState, prev: AppState): void {
    const changedFields = getChangedFields(next, prev)
    for (const listener of this.listeners) {
      listener(next, prev, changedFields)
    }
  }
}
```

## 19.6 渲染器实现

### 19.6.1 屏幕管理

```typescript
// ink/screen.ts
class Screen {
  private width: number
  private height: number
  private cursorX: number = 0
  private cursorY: number = 0
  private buffer: string[][] = []

  constructor() {
    const { columns, rows } = process.stdout
    this.width = columns
    this.height = rows
    this.clearBuffer()
  }

  clearBuffer(): void {
    this.buffer = Array(this.height)
      .fill(null)
      .map(() => Array(this.width).fill(' '))
  }

  write(x: number, y: number, text: string): void {
    for (let i = 0; i < text.length; i++) {
      if (x + i < this.width && y < this.height) {
        this.buffer[y][x + i] = text[i]
      }
    }
  }

  render(): void {
    // 使用 ANSI 转义序列移动光标并绘制
    process.stdout.write('\x1b[2J') // 清屏
    process.stdout.write('\x1b[H')  // 移动到左上角

    const output = this.buffer.map(row => row.join('')).join('\n')
    process.stdout.write(output)

    this.moveCursor(this.cursorX, this.cursorY)
  }

  moveCursor(x: number, y: number): void {
    process.stdout.write(`\1b[${y + 1};${x + 1}H`)
    this.cursorX = x
    this.cursorY = y
  }
}
```

```typescript
// 使用 Yoga 进行布局计算
import { Yoga, YogaNode } from 'yoga-js'

const computeLayout = (
  node: React.ReactElement
): LayoutResult => {
  const yogaNode = YogaNode.create()

  // 设置样式
  const style = node.props.style ?? {}

  if (style.width) yogaNode.setWidth(style.width)
  if (style.height) yogaNode.setHeight(style.height)
  if (style.flexDirection) {
    yogaNode.setFlexDirection(
      convertFlexDirection(style.flexDirection)
    )
  }
  if (style.justifyContent) {
    yogaNode.setJustifyContent(
      convertJustify(style.justifyContent)
    )
  }

  // 计算子节点
  for (const child of node.props.children ?? []) {
    const childNode = computeLayout(child)
    yogaNode.insertChild(childNode, yogaNode.getChildCount())
  }

  // 计算布局
  yogaNode.calculateLayout()

  return {
    left: yogaNode.getComputedLeft(),
    top: yogaNode.getComputedTop(),
    width: yogaNode.getComputedWidth(),
    height: yogaNode.getComputedHeight()
  }
}
```

### 19.7.2 协调循环

```typescript
// 协调循环
const performUnitOfWork = (fiber: Fiber): void => {
  // 1. 创建 DOM 节点
  const dom = createDOM(fiber)

  // 2. 为子节点创建 Fiber
  const children = fiber.props.children ?? []
  let prevSibling: Fiber | null = null

  for (const child of children) {
    const childFiber = createFiber(child, fiber)
    prevSibling = childFiber
  }

  // 3. 返回下一个工作单元
  if (fiber.child) {
    return fiber.child
  }

  // 4. 向上回溯
  let nextFiber = fiber
  while (nextFiber) {
    if (nextFiber.sibling) {
      return nextFiber.sibling
    }
    nextFiber = nextFiber.return
  }
}
```

## 19.9 事件处理

### 19.9.1 输入事件

```typescript
// ink/events/inputHandler.ts
class InputHandler {
  private keyHandlers: Map<string, KeyHandler> = new Map()

  constructor() {
    this.registerDefaultHandlers()
  }

  handleInput(data: Buffer): void {
    const key = parseKey(data)

    // 检查具体处理器
    const handler = this.keyHandlers.get(key.fullMatch)
    if (handler) {
      handler(key)
      return
    }

    // 通用处理器
    if (key.ctrl && key.name === 'c') {
      this.emit('interrupt')
    }
  }

  registerKeyHandler(
    key: string,
    handler: KeyHandler,
    options: { priority?: number } = {}
  ): void {
    this.keyHandlers.set(key, handler)
  }
}
```

### 19.9.2 焦点管理

```typescript
// ink/events/focusManager.ts
class FocusManager {
  private stack: string[] = []
  private currentIndex: number = 0

  push(id: string): void {
    this.stack.push(id)
    this.currentIndex = this.stack.length - 1
  }

  pop(): void {
    this.stack.pop()
    this.currentIndex = Math.max(0, this.stack.length - 1)
  }

  focusNext(): void {
    if (this.currentIndex < this.stack.length - 1) {
      this.currentIndex++
    }
  }

  focusPrev(): void {
    if (this.currentIndex > 0) {
      this.currentIndex--
    }
  }

  getCurrentFocus(): string | null {
    return this.stack[this.currentIndex] ?? null
  }
}
```

```typescript
const REPL: React.FC = () => {
```
